529 -def percentile(data, percent): 

530      """ 
531      Given an array of values and a percent value between 0 and 100, 
532      this function returns the percentile of the provided array.  For example,  
533      for percent=50 this will return the median value 
534   
535      This function is derived from a numerical recipe on the ActiveState web 
536      site, the link for whilch is available here: 
537   
538      http://code.activestate.com/recipes/511478-finding-the-percentile-of-the-values/ 
539   
540      We used this recipe so that we don't require all of scipy be installed, 
541      just for this one function.  Unlike the original recipe we sort the  
542      provided array we get so it doesn't need to be sorted ahead of time 
543   
544      Note that the value returned will always be rounded to the closest integer 
545      """ 
546   
547      sortedData = numpy.sort(data) 
548   
549      key   = (len(sortedData)-1) * (percent/100.0) 
550      lower = math.floor(key) 
551      upper = math.ceil(key) 
552   
553      try: 
554          if lower == upper:  
555              return sortedData[int(key)] 
556   
557          d0 = sortedData[int(lower)] * (upper - key) 
558          d1 = sortedData[int(upper)] * (key - lower) 
559          return int(round(d0 + d1)) 
560      except IndexError: 
561          return 0
