692          HTSeq.pair_SAM_alignments() function. 
693          """ 
694          if not read.paired_end: 
695              raise ValueError("%s is not a paired_end read" % repr(read))

697      def process_paired_reads(read1,read2):    
698          """ 
699          An enhanced implementation of the process_paired_reads() function 
700          embedded in the original HTSeq.pair_SAM_alignments() function.  This 
701          implementation removes some spurious warning messages, and also 
702          checks for the existance of certain object attributes such as  
703          iv.start_as_pos, before comparing them. 
704          """ 
705          if read1.pe_which == "second": 
706              aux = read1 
707              read1 = read2 
708              read2 = aux 
709          if not ((read1.pe_which == "first" and read2.pe_which == "second" ) or  
710                  (read1.pe_which == "unknown" and read2.pe_which == "unknown")): 
711              if verbose: 
712                  warnings.warn("Incorrect first/second assignments in mate" + 
713                                " pairs " + read1.read.name) 
714          if not ( read1.proper_pair and read2.proper_pair ): 
715              if verbose: 
716                  warnings.warn("Incorrect 'proper_pair' flag value for read" + 
717                                " pair " + read1.read.name) 
718          if hasattr(read2.iv, "start_as_pos"): 
719              if not (read1.mate_start == read2.iv.start_as_pos and  
720                      read2.mate_start == read1.iv.start_as_pos ): 
721                  if verbose: 
722                      warnings.warn("Read pair " + read1.read.name + 
723                                    " show inconsistency between 'iv'" + 
724                                    " and 'mate_start' values" ) 
725          return (read1, read2)

727      def process_single_read( read ): 
728          if read.mate_aligned:          
729              warnings.warn("Read " + read.read.name +  
730                            " claims to have an aligned mate which could not" + 
731                            " be found. (Is the SAM file properly sorted?)") 
732          if read.pe_which == "second": 
733              return (None, read) 
734          else: 
735              return (read, None)

737      def names_match(read1, read2): 
738          """Compare name of both records.  If it's a match, return True, all is good. 
739          If not however, compare the two strings again after stripping off any trailing 
740          _1 or _2 or /1 or /2 strings (from fastqkit).""" 
741   
742          if read1.read.name == read2.read.name: 
743              return True 
744          name1 = re.sub('[/_][12]$', '', read1.read.name) 
745          name2 = re.sub('[/_][12]$', '', read2.read.name) 
746          if name2 == name2: 
747              return True 
748          return False

751      alignments = iter(alignments) 
752      read1 = None 
753      read2 = None 
754      while True: 
755          if read1 is None: 
756              read1 = alignments.next() 
757              check_is_pe(read1) 
758          read2 = alignments.next() 
759          check_is_pe(read2)       
760          if names_match(read1, read2): 
761              yield process_paired_reads(read1, read2) 
762              read1 = None 
763              read2 = None 
764          else: 
765              yield process_single_read(read1) 
766              read1 = read2 
767              read2 = None
